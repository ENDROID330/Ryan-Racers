Car Ads

// --- Ads Logic --- const adsList = document.getElementById('adsList'); let ads = JSON.parse(localStorage.getItem('carAds')) || []; let currentUser = localStorage.getItem("currentUser") || "Guest"; function saveAds() { localStorage.setItem('carAds', JSON.stringify(ads)); } function renderAds(filteredAds) { adsList.innerHTML = ""; filteredAds.forEach((ad, index) => { const adDiv = document.createElement('div'); adDiv.classList.add('ad-card'); adDiv.innerHTML = `

${ad.title}

Price: $${ad.price}

${ad.description}

Uploaded by: ${ad.user} `; // Show edit/delete only if current user is uploader if (ad.user === currentUser) { const editBtn = document.createElement('button'); editBtn.textContent = "Edit"; editBtn.onclick = () => editAd(index); const deleteBtn = document.createElement('button'); deleteBtn.textContent = "Delete"; deleteBtn.onclick = () => deleteAd(index); adDiv.appendChild(editBtn); adDiv.appendChild(deleteBtn); } adsList.appendChild(adDiv); }); } function showAllAds() { renderAds(ads); } function showMyAds() { const myAds = ads.filter(ad => ad.user === currentUser); renderAds(myAds); } function editAd(index) { const ad = ads[index]; const newTitle = prompt("Edit title:", ad.title); const newPrice = prompt("Edit price:", ad.price); const newDesc = prompt("Edit description:", ad.description); if (newTitle) ad.title = newTitle; if (newPrice) ad.price = newPrice; if (newDesc) ad.description = newDesc; saveAds(); showAllAds(); } function deleteAd(index) { if (confirm("Are you sure you want to delete this ad?")) { ads.splice(index, 1); saveAds(); showAllAds(); } } // --- Firebase Auth State --- onAuthStateChanged(auth, (user) => { if (user) { currentUser = user.displayName || user.email; localStorage.setItem("currentUser", currentUser); console.log("Logged in as:", currentUser); } else { currentUser = "Guest"; localStorage.removeItem("currentUser"); console.log("No user logged in"); } showAllAds(); // refresh ads with correct user }); // --- Logout --- document.getElementById("logoutBtn").addEventListener("click", () => { signOut(auth).then(() => { localStorage.removeItem("currentUser"); currentUser = "Guest"; showAllAds(); console.log("Signed out"); }); }); // Default: show all ads showAllAds();
